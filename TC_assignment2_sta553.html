1.2 Only Adelie and Gentoo Penguins from Biscoe or Torgersen with Body Mass between 3500 and 5000

penguins2 <- na.omit(penguins) %>% 
  filter(species!="Chinstrap" & (island == 'Biscoe' | island == 'Torgersen') & 
           (body_mass_g < 5000 & body_mass_g > 3500)) %>%
  mutate(BMI = body_mass_g / 4000) %>%
  select(-X,-sex,-year,body_mass_g)

1.3 Create Scatter Plot of Flipper vs Bill Length

# isolate variables to be plotted
bill_length2<-penguins2$bill_length_mm
flipper_length2<-penguins2$flipper_length_mm
bmi <- penguins2$BMI

# find observation points for each species
species2<-penguins2$species

adelie.id2<-which(species2=='Adelie')
gentoo.id2<-which(species2=='Gentoo')

# create colors for points
col.code2 = c(alpha("red",0.5),alpha("blue",0.5))

# create blank plot
plot(bill_length2,flipper_length2, main="Flipper vs Bill Length",type="n")

# add points
points(bill_length2[adelie.id2], flipper_length2[adelie.id2], pch = 19, col = col.code2[1],cex=bmi[adelie.id2])
points(bill_length2[gentoo.id2], flipper_length2[gentoo.id2], pch = 19, col = col.code2[2],cex=bmi[gentoo.id2])

# add penguin pic
penguin_url<-"https://raw.githubusercontent.com/tcollier17/sta553/main/penguin.jpg"
z<-tempfile()
download.file(penguin_url,z,mode="wb")

library(jpeg)
pic<-readJPEG(z)
